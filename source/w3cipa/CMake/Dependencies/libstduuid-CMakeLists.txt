  GIT_REPOSITORY    https://github.com/mariusbancila/stduuid
  GIT_TAG           v1.2.3
  PREFIX            ${CMAKE_CURRENT_BINARY_DIR}/build
  CMAKE_ARGS        -DCMAKE_INSTALL_PREFIX=${OPEN_SRC_INSTALL_PREFIX} -DUUID_BUILD_TESTS=OFF -DUUID_TIME_GENERATOR=ON -DUUID_ENABLE_INSTALL=ON -DUUID_USING_CXX20_SPAN=OFF)
